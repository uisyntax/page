Logical Operators

    $var1: "panel";
    $var2: "rectangle";

    @mixin font-type(){
        &:after {
            @if $var1 == "panel" and $var2 == "square" {
                content: "var1 and var2 are true";
            }
            @else {
                content: "var1 and var2 are false";
            }
        }
    }



    h2 {
        @include font-type();
    }

    &=>Anything follows, add to parent

    css
    h2:after {
        content: "var1 and var2 are false";
    }
 
    Note: Instead of 'and', 'or' can be used
